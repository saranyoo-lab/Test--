<script>
let data = JSON.parse(localStorage.getItem("workData")) || {};

const today = new Date();
const todayStr = today.toISOString().slice(0,10);

document.getElementById("todayDate").innerText = todayStr;

function saveToday() {
  const count = parseInt(document.getElementById("count").value);
  if (!count) return alert("ใส่จำนวนก่อน");

  data[todayStr] = (data[todayStr] || 0) + count;
  localStorage.setItem("workData", JSON.stringify(data));

  renderAll();
}

function renderAll() {
  renderToday();
  renderSummary();
  renderCalendar();
}

function renderToday() {
  const total = data[todayStr] || 0;
  document.getElementById("todayTotal").innerText = `วันนี้ทำได้: ${total} ตัว`;
}

function renderSummary() {
  let daily = data[todayStr] || 0;
  let weekly = 0;
  let monthly = 0;

  const currentMonth = today.getMonth();

  Object.keys(data).forEach(date => {
    const d = new Date(date);
    const diff = (today - d) / (1000 * 60 * 60 * 24);

    if (diff <= 7) weekly += data[date];
    if (d.getMonth() === currentMonth) monthly += data[date];
  });

  document.getElementById("daily").innerText = `วันนี้: ${daily} ตัว`;
  document.getElementById("weekly").innerText = `สัปดาห์นี้: ${weekly} ตัว`;
  document.getElementById("monthly").innerText = `เดือนนี้: ${monthly} ตัว`;
}

function renderCalendar() {
  const calendar = document.getElementById("calendar");
  calendar.innerHTML = "";

  const year = today.getFullYear();
  const month = today.getMonth();

  const firstDay = new Date(year, month, 1).getDay();
  const daysInMonth = new Date(year, month + 1, 0).getDate();

  for (let i = 0; i < firstDay; i++) {
    calendar.innerHTML += `<div></div>`;
  }

  for (let d = 1; d <= daysInMonth; d++) {
    const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    const value = data[dateStr] || 0;

    calendar.innerHTML += `
      <div class="day ${dateStr === todayStr ? 'today' : ''}">
        <b>${d}</b><br>${value}
      </div>
    `;
  }
}

function showSection(section) {
  document.getElementById("summary").style.display = section === 'summary' ? 'block' : 'none';
  document.getElementById("calendarSection").style.display = section === 'calendar' ? 'block' : 'none';
}

renderAll();
</script>
